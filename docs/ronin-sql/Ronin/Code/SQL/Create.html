def emit
  tokens = emit_token('CREATE')
  tokens += emit_token('TEMP') if @temp

  tokens += emit_token(@type)

  tokens += emit_token('IF NOT EXISTS') if @if_not_exists
  tokens += emit_token(@name)

  return tokens + super
end
