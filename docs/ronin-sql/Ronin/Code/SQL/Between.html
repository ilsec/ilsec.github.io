def emit
  tokens = emit_value(@expr)

  tokens += emit_token('NOT') if @negated
  tokens += emit_token('BETWEEN')

  tokens += emit_value(@lower)
  tokens += emit_token('AND')
  tokens += emit_value(@higher)

  return tokens
end
