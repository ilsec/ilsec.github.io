def emit
  tokens = emit_value(@left)

  tokens += emit_token('NOT') if @negated

  tokens += emit_token(@op)
  tokens += emit_value(@right)

  if @escape
    tokens += emit_token('ESCAPE')
    tokens << @escape.to_s[0..0]
  end

  return tokens
end
