def =~(data)
  data = data.to_s

  @patterns.each do |pattern|
    if (index = (pattern =~ data))
      return index
    end
  end

  return nil
end

def match(data)
  data = data.to_s

  @patterns.each do |pattern|
    match = data.match(pattern)

    return Message.new(@name,@dialect,match[0]) if match
  end

  return nil
end
