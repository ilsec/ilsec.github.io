def address
  if env.has_key?('REMOTE_PORT')
    "#{ip}:#{env['REMOTE_PORT']}"
  else
    ip
  end
end

def headers
  headers = {}

  env.each do |name,value|
    if name =~ /^HTTP_/
      header_name = Network::HTTP.header_name(name.sub('HTTP_',''))
      headers[header_name] = value
    end
  end

  return headers
end
