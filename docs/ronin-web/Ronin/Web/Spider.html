def initialize(options={})
  options = {
    :proxy => Web.proxy,
    :user_agent => Web.user_agent
  }.merge(options)

  super(options)

  if options.fetch(:verbose,true)
    every_url do |url|
      print_info("Spidering #{url}")
    end
  end

  yield self if block_given?
end
