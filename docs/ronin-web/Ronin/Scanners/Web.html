def each_target(&block)
  print_info "Started web spidering ..."

  history.clear

  unless visit_hosts.empty?
    enqueue(URI::HTTP.build(:host => visit_hosts.first))
  end

  run do |page|
    print_info "Scanning page: #{page.url}"
    block.call(page)
  end

  print_info "Finished web spidering."
end
