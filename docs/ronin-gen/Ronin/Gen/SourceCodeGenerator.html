def self.template(name=nil)
  if name
    @template = name
  else
    @template ||= if superclass < SourceCodeGenerator
                    superclass.template
                  end
  end
end

def generate
  template self.class.template, @path

  if (edit? && editor?)
    # spawn the text editor for the newly generated file
    system(editor,@path)
  end
end
