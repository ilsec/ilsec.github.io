def authenticate(pop3,username,password)
  begin
    pop3.start(username,password)
  rescue Net::POPAuthenticationError
    return false
  end

  return true
end

def open_session
  pop3 = Net::POP3.new(self.host,self.port)
  pop3.enable_ssl if self.ssl

  return pop3
end
