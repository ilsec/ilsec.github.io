def scan!
  enum = if options.save?
           @scanner.save_each
         else
           @scanner.each
         end

  enum.each_with_index do |result,index|
    if result.kind_of?(DataMapper::Resource)
      print_resource(result)
    else
      print_result(result)
    end

    break if (options[:first] && index+1 == options[:first])
  end
end
