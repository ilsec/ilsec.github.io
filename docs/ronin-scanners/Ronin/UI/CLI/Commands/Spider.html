def execute
  Database.setup

  @scanner = Scanners::Spider.new
  @scanner.params = options
  @scanner.hosts = hosts

  print_info 'Saving spidered URLs ...' if options.save?

  scan!

  print_info 'All spidered URLs saved.' if options.save?
end

def print_result(page)
  print_info page.url

  if options.verbose?
    print_hash page.headers
    puts page.body
  end
end
