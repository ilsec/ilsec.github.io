def execute
  Database.setup

  @scanner = Scanners::SSH.new
  @scanner.host = host
  @scanner.params = options

  print_info 'Saving captured SSH credentials ...' if options.saved?

  scan!

  print_info 'All valid SSH credentials saved.' if options.save?
end
