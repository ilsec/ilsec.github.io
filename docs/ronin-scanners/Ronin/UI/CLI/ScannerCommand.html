def scan
  enum, printer = if import?
                    [scanner.import, method(:print_resource)]
                  else
                    [scanner.each, method(:print_result)]
                  end

  first_n = (@first || Float::INFINITY)
  count   = 0

  enum.each do |result|
    count += 1

    printer.call(result)

    break if count >= first_n
  end
end
