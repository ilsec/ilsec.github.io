def http_request(options={})
  options = Network::HTTP.expand_options(options)
  uri_class = if options[:ssl]
                URI::HTTPS
              else
                URI::HTTP
              end

  uri = uri_class.build(
    :host => options[:host],
    :port => options[:port],
  )
  request = Network::HTTP.request(options)

  return http.request(uri,request)
end
