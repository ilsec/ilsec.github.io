def scan
  super do |host|
    host.each_open_port do |open_port|
      type = case open_port.service
             when /SOCKS/i
               'socks'
             else # assume it's an HTTP proxy
               'http'
             end

      ip = new_ip(host)
      port = new_port(open_port)

      proxy = Proxy.first_or_new(:ip_address => ip, :port => port)
      proxy.type = type

      yield proxy if proxy.test
    end
  end
end
