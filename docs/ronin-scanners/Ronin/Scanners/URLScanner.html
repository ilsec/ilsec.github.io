def new_resource(result)

  new_url.last_scanned_at = Time.now
  return new_url
end

def normalize_result(result)
  unless result.kind_of?(::URI::Generic)
    begin
      URI.parse(result.to_s)
    rescue URI::InvalidURIError, URI::InvalidComponentError
    end
  else
    result
  end
end
