def +(obj)
  new_block = self.clone

  if obj.kind_of?(Block)
    new_block.elements += obj.elements
  else
    new_block.elements << obj
  end

  return new_block
end

def <<(obj)
  if obj.kind_of?(Array)
    obj.each { |element| self << element }
  else
    @elements << obj
  end

  return self
end
