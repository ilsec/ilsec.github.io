def ==(ins)
  return false unless @name==ins.name
  return false unless @suffix==ins.suffix

  return @args==ins.args
end

def compile
  unless @args.empty
    return "#{@name}#{@suffix}\t#{@args.join(', ')}"
  else
    return @name.to_s
  end
end
