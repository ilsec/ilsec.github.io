def comment_block(text)
  output = [comment('')]
  text.each_line { |line| output << comment(line) }
  output << comment('')

  return output.join($/)
end

def setup
  @comment = case options[:syntax]
            when 'intel'
              ';'
            when 'att'
              '#'
            end

  @metadata = YAML.dump({
    'arch' => options[:arch],
    'machine' => options[:machine],
    'syntax' => options[:syntax],
    'preproc' => options[:preproc]
  })
end
