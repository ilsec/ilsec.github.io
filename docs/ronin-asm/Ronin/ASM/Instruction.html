def initialize(name,operands)
  operands = operands.map do |op|
    case op
    when Integer, nil then ImmediateOperand.new(op)
    else                   op
    end
  end

  super(name,operands)
end

def width
  self.operands.map { |op|
    op.width if op.respond_to?(:width)
  }.compact.max
end
